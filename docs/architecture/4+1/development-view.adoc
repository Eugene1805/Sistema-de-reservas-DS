|===
| Pre-condiciones  | Post-condiciones
| context IHacerReservación::crearReservación(huespedId: int, datosDePago: DatosDePago, habitación: Habitación): void
  pre:
    -- Debe existir el huésped indicado por 'huespedId'.
    huespedes->exists(h \| h.idHuesped = huespedId) and

    -- La habitación debe existir y estar disponible.
    habitaciones->exists(hab \| hab.idHabitacion = habitación.idHabitacion and hab.estado = 'DISPONIBLE') and

    -- Los datos de pago deben ser válidos.
    datosDePago.esValido()

  context IHacerReservación::obtenerDatosHuésped(): Huesped
  pre:
    -- Debe existir una sesión activa asociada a un huésped autenticado.
    sesionesActivas->exists(s \| s.estado = 'ACTIVA')

  context IHacerReservación::obtenerDetallesHabitación(periodo: RangoDeFechas, habitación: Habitación): Habitación
  pre:
    -- El periodo debe ser válido y la habitación debe existir.
    periodo.fechaFin >= periodo.fechaInicio and
    habitaciones->exists(hab \| hab.idHabitacion = habitación.idHabitacion)

  context IHacerReservación::realizarPago(datosDePago: DatosDePago, monto: Divisa): void
  pre:
    -- El monto debe ser positivo y los datos de pago válidos.
    monto.cantidad > 0 and datosDePago.esValido()

  context IHacerReservación::enviarConfirmación(reservaId: int): void
  pre:
    -- Debe existir la reserva a confirmar.
    reservas->exists(r \| r.idReserva = reservaId)

| context IHacerReservación::crearReservación(huespedId: int, datosDePago: DatosDePago, habitación: Habitación): void
  post:
    -- Se ha creado una nueva reserva confirmada para el huésped y la habitación indicados.
    reservas->size = reservas@pre->size + 1 and
    let nuevaReserva : Reserva = reservas->excluding(reservas@pre)->asSequence()->first() in
    nuevaReserva.idReserva.oclIsDefined() and
    nuevaReserva.infoHuesped.oclIsNotEmpty() and
    nuevaReserva.estado = 'CONFIRMADA' and
    nuevaReserva.periodoDeEstancia.oclIsDefined()

  context IHacerReservación::obtenerDatosHuésped(): Huesped
  post:
    -- El resultado contiene los datos del huésped autenticado y no modifica el estado.
    result.oclIsTypeOf(Huesped) and
    huespedes = huespedes@pre

  context IHacerReservación::obtenerDetallesHabitación(periodo: RangoDeFechas, habitación: Habitación): Habitación
  post:
    -- El resultado corresponde a la habitación solicitada (operación de consulta).
    result.idHabitacion = habitación.idHabitacion and
    habitaciones = habitaciones@pre

  context IHacerReservación::realizarPago(datosDePago: DatosDePago, monto: Divisa): void
  post:
    -- Se ha registrado una transacción de pago aprobada por el monto indicado.
    transacciones->size = transacciones@pre->size + 1 and
    let nuevaTx : Transaccion = transacciones->excluding(transacciones@pre)->asSequence()->first() in
    nuevaTx.monto = monto and
    nuevaTx.estado = 'APROBADA'

  context IHacerReservación::enviarConfirmación(reservaId: int): void
  post:
    -- Se registró el envío de una confirmación sin alterar la cantidad de reservas.
    reservas->size = reservas@pre->size

    nuevoPago.metodo = TipoMétodoPago::TARJETA and
    nuevoPago.estado = 'APROBADO'

|===
| Pre-condiciones  | Post-condiciones
|context IHacerPagoConPayPal::iniciarTransacciónPayPal(tipo: TipoMétodoPago, monto: Divisa): void
  pre:
    -- El monto debe ser positivo y el tipo de método debe ser PAYPAL.
    monto.cantidad > 0 and tipo = TipoMétodoPago::PAYPAL

  context IHacerPagoConPayPal::confirmarTransacción(): void
  pre:
    -- Debe existir una transacción pendiente de confirmación para el actor actual.

.Diagrama de interfaces: Hacer pago con otro método
image::IHacerPagoConOtroMétodo.png[width=600,alt=Diagrama de interfaces de Hacer pago con otro método]

|===
| Pre-condiciones  | Post-condiciones
|context IHacerPagoConOtroMétodo::generarInstrucciónDePago(metodo: TipoMétodoPago, monto: Divisa): void
  pre:
    -- El monto debe ser positivo y el método de pago debe ser soportado (TRANSFERENCIA/OTRO).
    monto.cantidad > 0 and (metodo = TipoMétodoPago::TRANSFERENCIA or metodo = TipoMétodoPago::OTRO)
  context IMetodoPagoMgt::confirmarPagoAsincrono(referenciaPago: String): ResultadoTransaccion
  pre:
    -- Debe existir una transacción con la referencia proporcionada y debe estar esperando pago.
    transacciones->exists(tx \| tx.referencia = referenciaPago and tx.estado = 'ESPERANDO_PAGO')
|context IHacerPagoConOtroMétodo::generarInstrucciónDePago(metodo: TipoMétodoPago, monto: Divisa): void
  post:
    -- Se ha creado un nuevo registro de transacción con estado 'ESPERANDO_PAGO'.
    transacciones->size = transacciones@pre->size + 1 and
    let nuevaTx : Transaccion = transacciones->excluding(transacciones@pre)->asSequence()->first() in
    nuevaTx.estado = 'ESPERANDO_PAGO' and
    nuevaTx.monto = monto and
    (nuevaTx.metodo = TipoMétodoPago::TRANSFERENCIA or nuevaTx.metodo = TipoMétodoPago::OTRO)
  context IMetodoPagoMgt::confirmarPagoAsincrono(referenciaPago: String): ResultadoTransaccion
  post:
    -- El estado de la transacción se actualiza a 'APROBADA'.
    let txActualizada : Transaccion = transacciones->any(tx \| tx.referencia = referenciaPago) in
    txActualizada.estado = 'APROBADA' and

    -- El resultado de la operación confirma el éxito.
    result.exitoso = true and
    result.idTransaccion = txActualizada.id

.Diagrama de interfaces: Cancelar reservación
image::ICancelarReservación.png[width=600,alt=Diagrama de interfaces de Cancelar reservación]
|===
| Pre-condiciones  | Post-condiciones
| context IMetodoPagoMgt::procesarReembolso(pagoId: String, monto: Divisa): ResultadoTransaccion
  pre:
    -- Debe existir un pago original aprobado con el ID proporcionado.
    pagos->exists(p \| p.id = pagoId and p.estado = 'APROBADO') and

    -- El monto a reembolsar no puede ser mayor al monto del pago original.
    monto <= pagos->any(p \| p.id = pagoId).monto
  context IConsultarPolitica::obtenerPoliticaCancelacion(hotelId: String, fecha: Date): PoliticaCancelacion
  pre:
    -- Debe existir un hotel con el ID proporcionado.
    hoteles->exists(h \| h.id = hotelId)
| context IMetodoPagoMgt::procesarReembolso(pagoId: String, monto: Divisa): ResultadoTransaccion
  post:
    -- Se ha creado una nueva transacción de tipo 'REEMBOLSO'.
    transacciones->size = transacciones@pre->size + 1 and
    let nuevaTx : Transaccion = transacciones->excluding(transacciones@pre)->asSequence()->first() in
    nuevaTx.tipo = 'REEMBOLSO' and
    nuevaTx.estado = 'APROBADA' and
    nuevaTx.pagoOriginalId = pagoId and

    -- El resultado confirma la operación exitosa.
    result.exitoso = true and
    result.idTransaccion = nuevaTx.id
  context IConsultarPolitica::obtenerPoliticaCancelacion(hotelId: String, fecha: Date): PoliticaCancelacion
  post:
    -- El resultado ('result') es la política vigente para el hotel en la fecha dada.
    result = politicas->select(p \| p.hotelId = hotelId and fecha.isBetween(p.fechaInicio, p.fechaFin))->asSequence()->first() and

    -- La operación es de solo lectura, no modifica el estado de las políticas.
    politicas = politicas@pre

|===
| Pre-condiciones  | Post-condiciones
|context IRegistrarConsumoExtra::agregarCargo(cargo: Consumo): void
  pre:
    -- Debe existir una cuenta/estancia activa asociada al consumo y el monto debe ser positivo.
    cuentas->exists(c \| c.idAsociado = cargo.estanciaId and c.estado = 'ABIERTA') and
    cargo.monto > 0
  context IRegistrarConsumoExtra::obtenerDetalles(): Consumo
  pre:
    -- Operación de consulta sin precondiciones estrictas.
    true
|context IRegistrarConsumoExtra::agregarCargo(cargo: Consumo): void
  post:
    -- La cuenta asociada a la estancia tiene un nuevo cargo registrado.
    let cuenta : Cuenta = cuentas->any(c \| c.idAsociado = cargo.estanciaId) in
    cuenta.cargos->size = cuenta.cargos@pre->size + 1 and
    cuenta.cargos->includes(cargo) and

    -- El saldo de la cuenta se ha incrementado por el monto del cargo.
    cuenta.saldo = cuenta.saldo@pre + cargo.monto and

    -- No hay valor de retorno; la operación actualiza el estado de la cuenta.
    true
  context IRegistrarConsumoExtra::obtenerDetalles(): Consumo
  post:
    -- Se devuelve un objeto Consumo con información detallada (operación de solo lectura).
    result.oclIsTypeOf(Consumo)

.Diagrama de interfaces: Iniciar sesión
image::IIniciarSesión.png[width=600,alt=Diagrama de interfaces de Iniciar sesión]
|===
| Pre-condiciones  | Post-condiciones
| context IIniciarSesión::validarCredenciales(credenciales: Credenciales): void
  pre:
    -- Las credenciales proporcionadas deben tener un formato válido (ej. email no vacío, contraseña no vacía).
    credenciales.usuario.oclIsNotEmpty() and credenciales.password.oclIsNotEmpty()
| context IIniciarSesión::validarCredenciales(credenciales: Credenciales): void
  post:
  -- Escenario de éxito: si las credenciales son correctas, se crea una nueva sesión.
  if actores->exists(a \| a.usuario = credenciales.usuario and a.passwordHash = encriptar(credenciales.password)) then
    sesionesActivas->size = sesionesActivas@pre->size + 1 and
    let nuevaSesion : Sesion = sesionesActivas->excluding(sesionesActivas@pre)->asSequence()->first() in
    nuevaSesion.actorId = actores->any(a \| a.usuario = credenciales.usuario).id and
    nuevaSesion.token.oclIsNotEmpty()
  else
    -- Escenario de falla: no se crea ninguna sesión.
    sesionesActivas->size = sesionesActivas@pre->size
  endif

|===
| Pre-condiciones  | Post-condiciones
|context IRegistrarPolítica::registrarPolítica(politica: Política): void
  pre:
    -- Los datos de la política deben ser consistentes y completos.
    politica.nombre.oclIsNotEmpty() and
    politica.vigencia.fechaFin >= politica.vigencia.fechaInicio
|context IRegistrarPolítica::registrarPolítica(politica: Política): void
  post:
    -- Se ha creado un nuevo objeto Politica en el sistema.
    politicas->size = politicas@pre->size + 1 and

    -- La nueva política es la que se acaba de agregar al conjunto.
    let nuevaPolitica : Politica = politicas->excluding(politicas@pre)->asSequence()->first() in

    -- Los atributos de la nueva política corresponden a los datos de entrada.
    nuevaPolitica.nombre = politica.nombre and
    nuevaPolitica.descripcion = politica.descripcion and
    nuevaPolitica.vigencia = politica.vigencia and
    nuevaPolitica.idPolitica.oclIsDefined()

|===
| Pre-condiciones  | Post-condiciones
| context IActualizarPolítica::actualizarPolítica(politica: Política, id: int): void
  pre:
    -- La política que se desea actualizar debe existir en el sistema y los nuevos datos deben ser consistentes.
    politicas->exists(p \| p.idPolitica = id) and

    let politicaActualizada : Politica = politicas->any(p \| p.idPolitica = id) in
    politicaActualizada.nombre = politica.nombre and
    politicaActualizada.descripcion = politica.descripcion and
    politicaActualizada.vigencia = politica.vigencia
